import { createServerClient } from '@supabase/ssr'
import { cookies } from 'next/headers'

export async function createClient() {
  const cookieStore = await cookies()
  return createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() { return cookieStore.getAll() },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value, options }) =>
            cookieStore.set(name, value, options)
          )
        }
      }
    }
  )
}

lib/supabase/client.ts - For Client Components

import { createBrowserClient } from '@supabase/ssr'

-- Create waitlist table
CREATE TABLE IF NOT EXISTS waitlist (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  email TEXT NOT NULL UNIQUE,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Add indexes
CREATE INDEX idx_waitlist_email ON waitlist(email);
CREATE INDEX idx_waitlist_created_at ON waitlist(created_at DESC);

-- Enable RLS
ALTER TABLE waitlist ENABLE ROW LEVEL SECURITY;

-- Allow anyone to insert
CREATE POLICY "Anyone can sign up for waitlist" ON waitlist
  FOR INSERT
  WITH CHECK (true);

-- Only authenticated users can view
CREATE POLICY "Authenticated users can view waitlist" ON waitlist
  FOR SELECT
  USING (auth.role() = 'authenticated');

5. Email Signup Troubleshooting

The Problem

❌ Error: "Failed to fetch" when submitting email signup form

Debugging Process

1 Initial Error Investigation

Browser console showed:

Error: TypeError: Failed to execute 'set' on 'Headers': Invalid value

This indicated an issue with HTTP headers in the Supabase client

2 Added Environment Variable Validation

Modified lib/supabase/client.ts:

export function createClient() {
  const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL
  const supabaseAnonKey = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY

  if (!supabaseUrl || !supabaseAnonKey) {
    throw new Error(
      'Missing Supabase environment variables. ' +
      `URL: ${supabaseUrl ? 'set' : 'missing'}, ` +
      `Key: ${supabaseAnonKey ? 'set' : 'missing'}`
    )
  }

  return createBrowserClient(supabaseUrl, supabaseAnonKey)
}
3 Created Debug Endpoint

File: app/debug-env/page.tsx

4 Created API Test Endpoint

File: app/api/test-waitlist/route.ts

5 Root Cause Found!
The Problem: Environment variables had trailing newline characters!

https://weivmnsonkkocnsovnjw.supabase.co\n
eyJhbGc...FxFpfqw\n

These newlines were causing the HTTP Header validation to fail.
6 Why Did This Happen?

When using Vercel CLI with printf:

printf "value\nn\n" | vercel env add KEY production

The \n characters were being included in the stored value!

7 The Solution

Method 1: Use Vercel Dashboard (RECOMMENDED)

  1. Go to Vercel Project → Settings → Environment Variables
  2. Delete the corrupted variables
  3. Add them manually through the web interface
  4. Paste values carefully - no extra spaces or newlines
  5. Redeploy

Method 2: Use printf without newline

printf "value" | vercel env add KEY production

But dashboard is more reliable!

💡 Critical Lessons:
  • Always use Vercel Dashboard for environment variables - CLI can introduce hidden characters
  • Create debug endpoints early - They save hours of troubleshooting
  • Test server-side and client-side separately - Helps isolate issues
  • Log everything during debugging - Console logs revealed the newline issue
  • Environment variable issues are silent killers - Build succeeds but runtime fails

// Server-side callback to exchange OAuth code for session
export async function GET(request: Request) {
  const code = requestUrl.searchParams.get('code')

  if (code) {
    const supabase = await createClient()
    await supabase.auth.exchangeCodeForSession(code)
  }

  return NextResponse.redirect(`${origin}/dashboard`)
}
